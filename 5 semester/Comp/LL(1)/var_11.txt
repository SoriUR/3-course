<последовательность_операторов_описания_данных> ::= <оператор_описания_данных> | <последовательность_операторов_описания_данных> <оператор_описания_данных>
<оператор_описания_данных> ::= <описатель> <список_идентификаторов>
<описатель> ::= "int" | "float" | "time"
<список_идентификаторов> ::= "идентификатор" | <список_идентификаторов> "," "идентификатор"
<имя_функции> ::= "идентификатор"

<блок> ::= "{" <последовательность_операторов> "}"
<последовательность_операторов> ::= <оператор> | <последовательность_операторов> <оператор>
